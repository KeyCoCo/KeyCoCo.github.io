mySql基础语法

连接数据库命令：mysql -uroot -padmin;

查看已有数据库：show databases;

创建一个库：create database 库名;

选择一个库：use 库名;

查看库内的数据表：show tables;

在库内创建一个数据表：create table 表名(字段名 字段类型 ，字段名 字段类型 ，。。。。。)charset=utf8;

create table MyTable(
MyTable_Id int not null auto_increment,
MyTable_Data varchar(20),
rimmary key (MyTable_Id)
)charset=utf8;

在表内插入数据：insert into 表名 (字段名)values(值);

在表内删除数据：delete from 表名 where 筛选数据;

在表内修改数据：update 表名 set 字段名 = 修改的值,字段名 = 修改的值where 筛选数据;

查看表中所有数据：select * from 表名;

在表内插入新的字段：alert table 表名 add 字段名 字段属性;

在表内删除字段：alert table 表名 drop 字段名;
